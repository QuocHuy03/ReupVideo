# PARAM music type=file label="Nhạc nền" filter="Audio (*.mp3 *.wav *.aac *.m4a)"
# PARAM bgm_volume type=number label="Âm lượng BGM" default=0.2 min=0 max=2 step=0.05
# PARAM output_volume type=number label="Âm lượng tổng" default=1.0 min=0 max=2 step=0.05
ffmpeg -y -i {input} -i {music} -filter_complex "[1:a]volume={bgm_volume}[bgm];[0:a][bgm]amix=inputs=2:duration=first:dropout_transition=2,volume={output_volume}" -c:v copy -c:a aac -shortest {output_mp4}
